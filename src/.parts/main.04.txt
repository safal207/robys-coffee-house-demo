  const year = qs<HTMLElement>("#current-year");

  injectMarquee();
  injectLiveStatus();
  injectWorldExperience();
  injectStories();
  injectGallery();
  injectReviews();
  injectPassport();
  injectMap();
  injectMobileDock();
  injectMatcherDialog();

  setupLanguage();
  setupNavigation();
  setupHeaderAndProgress();
  setupRitualCards();
  setupGallery();
  setupPassport();
  setupMap();
  setupInstallPrompt();
  addHeroFallback();
  setupReveal();
  setupServiceWorker();

}
